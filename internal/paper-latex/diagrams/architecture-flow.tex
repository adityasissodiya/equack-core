\begin{tikzpicture}[
    every node/.style={font=\footnotesize\sffamily},
    stage/.style={
        rectangle,
        rounded corners=5pt,
        minimum width=3.0cm,
        minimum height=1.25cm,
        align=center,
        draw=#1!70!black,
        fill=#1!12,
        line width=0.8pt
    },
    side/.style={
        rectangle,
        rounded corners=4pt,
        minimum width=2.6cm,
        minimum height=0.9cm,
        align=center,
        draw=#1!65!black,
        fill=#1!10,
        line width=0.65pt
    },
    claim/.style={
        rectangle,
        rounded corners=5pt,
        align=center,
        draw=black!20,
        fill=black!3,
        line width=0.6pt,
        inner sep=6pt
    },
    flow/.style={
        -{Stealth[length=2.6mm, width=2.1mm]},
        line width=0.9pt,
        draw=black!65
    },
    aux/.style={
        -{Stealth[length=2.3mm, width=1.9mm]},
        dashed,
        line width=0.75pt,
        draw=black!55
    },
    note/.style={font=\scriptsize\sffamily, text=black!70, align=center}
]

\node[stage=figingest] (s1) at (0,0)
    {\textbf{1. Ingest signed event}\\[-1pt]
     verify signature, canonicalize, compute \texttt{op\_id}};

\node[stage=figdag] (s2) at (3.7,0)
    {\textbf{2. Activate causal DAG}\\[-1pt]
     stage pending parents, replay only ready events};

\node[stage=figtrust] (s3) at (7.4,0)
    {\textbf{3. Rebuild TrustView}\\[-1pt]
     issuer keys, status lists, VC verification context};

\node[stage=figgate] (s4) at (11.1,0)
    {\textbf{4. Build epochs + Gate}\\[-1pt]
     deterministic \texttt{GRANT}/\texttt{REVOKE}, deny-wins apply/skip};

\node[stage=figstate] (s5) at (14.8,0)
    {\textbf{5. Materialize CRDT state}\\[-1pt]
     apply authorized \texttt{DATA}; update digest and counters};

\node[stage=figquery] (s6) at (18.5,0)
    {\textbf{6. Query / projection}\\[-1pt]
     decrypt if authorized, otherwise redact};

\draw[flow] (s1) -- (s2);
\draw[flow] (s2) -- (s3);
\draw[flow] (s3) -- (s4);
\draw[flow] (s4) -- (s5);
\draw[flow] (s5) -- (s6);

\node[side=figsync] (remote) at (-0.9,1.85)
    {\textbf{remote sync}\\announce, fetch, parent repair};
\node[side=figingest] (local) at (1.1,1.85)
    {\textbf{local submit}\\PEP emits signed event};
\draw[aux, draw=figsync!80!black] (remote.south) -- ([xshift=-8pt]s1.north);
\draw[aux, draw=figingest!80!black] (local.south) -- ([xshift=8pt]s1.north);

\node[side=figstore, minimum width=4.8cm] (store) at (7.4,2.05)
    {\textbf{append-only event store}\\ops, edges, checkpoints, trust evidence};
\draw[aux, draw=figstore!80!black] (s1.north) -- ++(0,0.55) -| (store.west);
\draw[aux, draw=figstore!80!black] (store.south) -- (s2.north);
\draw[aux, draw=figstore!80!black] ([xshift=10pt]store.south) -- ([xshift=-10pt]s5.north);

\node[side=figaudit, minimum width=4.8cm] (audit) at (11.1,-1.95)
    {\textbf{hash-chained audit}\\\texttt{AppliedOp} / \texttt{SkippedOp(reason)}};
\draw[aux, draw=figaudit!80!black] (s4.south) -- (audit.north);
\draw[aux, draw=figaudit!80!black] (s6.south) |- ([xshift=18pt]audit.north);

\node[claim, minimum width=18.8cm] (claimband) at (9.25,-3.25)
    {\textbf{Deterministic convergence invariant:} same reconciled event set $\Rightarrow$ same topo order $\Rightarrow$ same gate decisions $\Rightarrow$ same final state digest.};

\node[note, anchor=west] at (0.0,-4.1)
    {Replay semantics are primary; storage, networking, and audit are support paths rather than co-equal claims.};

\end{tikzpicture}
