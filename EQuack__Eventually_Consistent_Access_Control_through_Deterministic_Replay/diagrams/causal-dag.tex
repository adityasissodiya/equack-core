    },

        fill=#1!20,
        rounded corners=2pt,

    replaynum/.style={
        circle,
        fill=black!80,
        text=white,
        font=\scriptsize\sffamily\bfseries,
        inner sep=1.5pt,
        minimum size=5mm
    },
    %% --- Annotation box ---
    annot/.style={
        rectangle,
        rounded corners=4pt,
        draw=gray!60,
        fill=white,
        align=center,
        font=\scriptsize\sffamily,
        inner sep=5pt,
        line width=0.6pt
    }
]
