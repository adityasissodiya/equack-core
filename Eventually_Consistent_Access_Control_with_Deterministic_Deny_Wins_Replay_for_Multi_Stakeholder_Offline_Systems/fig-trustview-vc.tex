% ECAC TrustView & VC Verification Flow
% Figure 4: For Section 3.2 (VCs, TrustView)

\begin{tikzpicture}[
    % Styles
    opnode/.style={
        rectangle, rounded corners=4pt,
        minimum width=2.2cm, minimum height=0.8cm,
        font=\tiny\sffamily\bfseries, align=center,
        draw=#1!70!black, fill=#1!25, line width=0.6pt
    },
    databox/.style={
        rectangle, rounded corners=5pt,
        font=\tiny\sffamily, align=left,
        draw=black!50, fill=white, line width=0.5pt,
        inner sep=6pt
    },
    process/.style={
        rectangle, rounded corners=4pt,
        minimum width=2.8cm, minimum height=0.9cm,
        font=\scriptsize\sffamily, align=center,
        draw=#1!60!black, fill=#1!15, line width=0.6pt
    },
    arrow/.style={
        -{Stealth[length=2mm, width=1.5mm]},
        line width=0.7pt, draw=black!60
    },
    zone/.style={
        rectangle, rounded corners=8pt,
        draw=black!40, fill=black!3, line width=0.8pt,
        inner sep=8pt
    },
    zonetitle/.style={
        font=\small\sffamily\bfseries, text=black!70
    }
]

% Colors
\definecolor{trust}{HTML}{9B59B6}
\definecolor{verify}{HTML}{3498DB}
\definecolor{success}{HTML}{27AE60}

% === LEFT SIDE: IN-BAND TRUST ASSEMBLY ===
\node[zonetitle] at (3.5, 9) {IN-BAND TRUST ASSEMBLY};
\node[font=\tiny\sffamily, text=black!50] at (3.5, 8.5) {(from ops in topo order)};

% Input ops
\node[opnode=trust] (issuerkey) at (1, 7.5) {IssuerKey};
\node[font=\tiny\sffamily, text=trust!70!black] at (2.8, 7.5) {first-wins};

\node[opnode=trust] (issuerrev) at (1, 6.5) {IssuerKeyRevoke};
\node[font=\tiny\sffamily, text=trust!70!black] at (3.1, 6.5) {sets revoked\_at};

\node[opnode=trust] (statuschunk) at (1, 5.5) {StatusListChunk};
\node[font=\tiny\sffamily, text=trust!70!black] at (3.1, 5.5) {latest version};

% TrustView box
\node[zone, minimum width=5.5cm, minimum height=5cm] (tvzone) at (3.5, 2.8) {};
\node[zonetitle, anchor=north] at (3.5, 5.1) {TrustView};

% Issuer keys structure
\node[databox, minimum width=4.8cm, anchor=north] (ikbox) at (3.5, 4.6) {
    \textbf{issuer\_keys:}\\
    \quad issuer\_id $\rightarrow$ key\_id $\rightarrow$\\
    \quad\quad IssuerKeyRecord \{\\
    \quad\quad\quad pubkey: [u8; 32]\\
    \quad\quad\quad valid\_from\_ms\\
    \quad\quad\quad valid\_until\_ms\\
    \quad\quad\quad activated\_at\_ms\\
    \quad\quad\quad revoked\_at\_ms\\
    \quad\quad \}
};

% Status lists structure
\node[databox, minimum width=4.8cm, anchor=north] (slbox) at (3.5, 1.8) {
    \textbf{status\_lists:}\\
    \quad list\_id $\rightarrow$ StatusList \{\\
    \quad\quad issuer\_id\\
    \quad\quad version\\
    \quad\quad chunks: BTreeMap\\
    \quad\quad bitset\_sha256\\
    \quad \}
};

% Arrows from ops to TrustView
\draw[arrow, color=trust!70!black] (issuerkey.east) -- ++(1.5, 0) |- ($(ikbox.north west)+(0.5,-0.3)$);
\draw[arrow, color=trust!70!black] (issuerrev.east) -- ++(1.0, 0) |- ($(ikbox.west)+(0,-0.5)$);
\draw[arrow, color=trust!70!black] (statuschunk.east) -- ++(0.5, 0) |- (slbox.west);

% === RIGHT SIDE: VC VERIFICATION FLOW ===
\node[zonetitle] at (11, 9) {VC VERIFICATION FLOW};

% JWT input
\node[databox, minimum width=3.5cm] (jwt) at (11, 7.8) {
    \textbf{Compact JWT}\\
    header.payload.signature\\
    \tiny (base64url encoded)
};

% Step 1: Parse
\node[process=verify] (parse) at (11, 6.4) {Parse JWS\\(3 parts)};
\draw[arrow] (jwt) -- (parse);

% Step 2: Extract issuer
\node[process=verify] (extract) at (11, 5.2) {Extract issuer\\from payload};
\draw[arrow] (parse) -- (extract);

% Step 3: Lookup key
\node[process=verify] (lookup) at (11, 4) {Lookup key in\\TrustView};
\draw[arrow] (extract) -- (lookup);

% Arrow from TrustView to lookup
\draw[arrow, dashed, color=trust!70!black] (tvzone.east) -- ++(1, 0) |- (lookup.west);

% Step 4: Verify sig
\node[process=verify] (verifysig) at (11, 2.8) {Ed25519 verify\\(header.payload)};
\draw[arrow] (lookup) -- (verifysig);

% Step 5: Check status
\node[process=verify] (checkstatus) at (11, 1.6) {Check status list\\revocation};
\draw[arrow] (verifysig) -- (checkstatus);

% Arrow from status lists
\draw[arrow, dashed, color=trust!70!black] (slbox.east) -- ++(2, 0) |- (checkstatus.west);

% Step 6: Extract claims
\node[process=success] (claims) at (11, 0.4) {Extract claims\\role, scope, nbf, exp};
\draw[arrow] (checkstatus) -- (claims);

% Output: VerifiedVc
\node[databox, minimum width=4cm, fill=success!10] (vvc) at (11, -1.2) {
    \textbf{VerifiedVc} \{\\
    \quad cred\_hash: blake3(jwt)\\
    \quad subject\_pk\\
    \quad role, scope\_tags\\
    \quad nbf\_ms, exp\_ms\\
    \}
};
\draw[arrow, color=success!70!black, line width=1pt] (claims) -- (vvc);

% Connection note
\node[font=\tiny\sffamily, text=black!50, align=center] at (7.2, 0) {
    TrustView built\\
    from log ops\\
    (no filesystem)
};

\end{tikzpicture}
